import sys

from collections import deque
#deque이용하면 list의경우 맨앞에 팝할경우 할때마다 앞으로 한칸씩 이동하므로
#비효율적이다 그러므로 deque사용 deque는 이동 x
#n은 승객의 수 m은 최대 몸무게 두명이하의
n,m =map(int,input().split())

a=list(map(int,input().split()))
a.sort()
a=deque(a)


rt=0
lt=n-1

cnt=0

while a:
    if len(a) ==1:
        cnt+=1
        break
    if a[0]+a[-1]>m:
        a.pop()
        cnt+=1
    else:
        a.popleft()
        a.pop()
        cnt+=1
print(cnt)
